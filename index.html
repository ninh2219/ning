Kho lưu trữ các gói .deb dành cho môi trường Linux (Debian/Ubuntu).

Hướng dẫn cài đặt

Mở Terminal và chạy các lệnh dưới đây để thêm repo này vào hệ thống của bạn.

# 1. Thêm GPG Key để xác thực repo
curl -s --compressed "https://ninh2219.github.io/ning/KEY.gpg" | gpg --dearmor | sudo tee /etc/apt/trusted.gpg.d/ninh_repo.gpg >/dev/null

# 2. Thêm repo vào danh sách nguồn
echo "deb [arch=amd64 signed-by=/etc/apt/trusted.gpg.d/ninh_repo.gpg] https://ninh2219.github.io/ning/repo stable main" | sudo tee /etc/apt/sources.list.d/ninh_repo.list

# 3. Cập nhật và cài đặt
sudo apt update
# sudo apt install <tên-gói>
